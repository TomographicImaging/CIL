        self.fp = AstraForwardProjectorMC(volume_geometry=geomv,

[docs]    def direct(self, IM, out=None):
        self.fp.set_input(IM)
        
        if out is None:
            return self.fp.get_output()
        else:
            out.fill(self.fp.get_output())
